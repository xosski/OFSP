    print(f"✓ Available providers: {providers}")
    
    # Test with fallback (always works)
    response = ai.llm_chat("What is security?", provider="fallback")
    print(f"✓ Fallback response: {response[:100]}...")
    
    # Test with first available provider
    if providers:
        provider = providers[0]
        try:
            response = ai.llm_chat("Explain XSS", provider=provider)

db_path = 'test_verify.db'
